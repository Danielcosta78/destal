<script src="https://unpkg.com/peerjs@1.4.7/dist/peerjs.min.js"></script>
<script>
  const peer = new Peer(undefined, {
    host: 'servidor-pn0h.onrender.com',
    port: 443,
    path: '/peerjs',
    secure: true
  });

  peer.on('open', id => {
    console.log('Conectado ao servidor PeerJS personalizado. Meu ID:', id);
  });

  peer.on('error', err => {
    console.error('Erro ao conectar ao servidor PeerJS:', err);
  });
</script>
